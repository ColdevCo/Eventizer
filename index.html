Custom Hook

eventizer_init
Triggered after Eventizer initialized
add_action( 'eventizer_init', 'your_callback_here' );
function your_callback_here() {
    // Your code here
}
eventizer_render
Triggered after Eventizer render event's form
add_action( 'eventizer_render', 'your_callback_here' );
function your_callback_here() {
    // Your code here
}
eventizer_save
Triggered after Eventizer save event's data
add_action( 'eventizer_save', 'your_callback_here' );
function your_callback_here() {
    // Your code here
}

Custom Filter

ev_setting_tabs
To add tab menu items inside Eventizer Settings
add_action( 'ev_setting_tabs', 'your_callback_here' );
function your_callback_here( $tab_items ) {
    // Your code here
    return $tab_items;
}

Function

add_event_options
To add Event Option
add_event_options( 'option_key', 'option_value' );
get_event_options
To get Event Option
get_event_options( 'option_key' );
update_event_options
To update Event Option
update_event_options( 'option_key', 'option_value' );
delete_event_options
To delete Event Option
delete_event_options( 'option_key' );
add_mail
To add Event Mail
$mail = array( 
    'cem_mail_editor_subject' => 'your_mail_subject',
    'cem_mail_editor_content' => 'your_mail_content',
    'cem_mail_editor_context' => 'your_mail_context'
);
add_mail( $mail );
update_mail
To update Event Mail
$mail_id = // Your mail ID
$mail = array( 
    'cem_mail_editor_subject' => 'your_mail_subject',
    'cem_mail_editor_content' => 'your_mail_content',
    'cem_mail_editor_context' => 'your_mail_context'
);
update_mail( $mail_id, $mail );
delete_mail
To delete Event Mail
$mail_context = // Your mail context
delete_mail( $mail_context );
get_all_mails
To get all Event Mails
get_all_mails();
get_mail
To get Event Mail by id
$mail_id = // Your mail ID
get_mail( $mail_id );
get_mail_by_context
To get Event Mail by context
$mail_context = // Your mail context
get_mail_by_context( $mail_context );
